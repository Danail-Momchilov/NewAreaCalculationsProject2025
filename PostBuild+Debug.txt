Post-Build (add to .csproj):
==============================

1. Add this property inside <PropertyGroup>:
    <AddinDir>$(AppData)\Autodesk\Revit\Addins\2026\</AddinDir>

2. Add this ItemGroup after </PropertyGroup>:
  <ItemGroup>
    <AddinImages Include="$(ProjectDir)img\*.png" />
    <AddinManifest Include="$(ProjectDir)manifest.addin" />
  </ItemGroup>

3. Add this Target:
  <Target Name="CopyToRevitAddins" AfterTargets="Build">
    <ItemGroup>
      <OutputDlls Include="$(TargetDir)*.dll" />
      <OutputJson Include="$(TargetDir)*.deps.json" />
    </ItemGroup>
    <Copy SourceFiles="@(AddinManifest)" DestinationFolder="$(AddinDir)" SkipUnchangedFiles="true" />
    <Copy SourceFiles="@(AddinImages)" DestinationFolder="$(AddinDir)" SkipUnchangedFiles="true" />
    <Copy SourceFiles="@(OutputDlls)" DestinationFolder="$(AddinDir)" SkipUnchangedFiles="true" />
    <Copy SourceFiles="@(OutputJson)" DestinationFolder="$(AddinDir)" SkipUnchangedFiles="true" />
  </Target>
